async function verificarSQL() { const btn = document.getElementById("btn-enviar"); btn.disabled = true; btn.textContent = "⏳ Executando..."; const outputEl = document.getElementById("saida-sql"); outputEl.style.display = "block"; outputEl.textContent = ""; try { if (!db) { const SQL = await initSqlJs({ locateFile: file => `https://sql.js.org/dist/${file}` }); db = new SQL.Database(); } const fase = listaDesafios[indiceFaseAtual]; // Recria o banco a partir do setup db.exec("DROP TABLE IF EXISTS alunos;"); db.exec(fase.sqlSetup); let codigoUsuario = editorSQL.getValue().trim(); if (!codigoUsuario) { alert("Por favor, escreva uma consulta SQL."); btn.disabled = false; btn.textContent = "⚡ Executar consulta"; return; } // Remove ponto e vírgula final para evitar erro no SQL.js if (codigoUsuario.endsWith(';')) { codigoUsuario = codigoUsuario.slice(0, -1); } let resultadoUsuario; try { resultadoUsuario = db.exec(codigoUsuario); } catch (e) { outputEl.textContent = "❌ Erro na sua consulta:\n" + e.message; btn.disabled = false; btn.textContent = "⚡ Executar consulta"; return; } // Garantir que resultadoUsuario tenha pelo menos um item if (!resultadoUsuario || resultadoUsuario.length === 0) { outputEl.textContent = "⚠️ Sua consulta não retornou resultados."; btn.disabled = false; btn.textContent = "⚡ Executar consulta"; return; } // Extrair colunas e valores do usuário const colsUsuario = resultadoUsuario[0].columns; const valsUsuario = resultadoUsuario[0].values; // Executar gabarito let gabaritoQuery = fase.sqlGabarito.trim(); if (gabaritoQuery.endsWith(';')) { gabaritoQuery = gabaritoQuery.slice(0, -1); } let resultadoGabarito; try { resultadoGabarito = db.exec(gabaritoQuery); } catch (e) { outputEl.textContent = "❌ Erro no gabarito (contate o professor)."; btn.disabled = false; btn.textContent = "⚡ Executar consulta"; return; } const valsGabarito = resultadoGabarito[0].values; // --- COMPARAÇÃO APENAS DOS VALORES --- const iguais = JSON.stringify(valsUsuario) === JSON.stringify(valsGabarito); // --- EXIBIÇÃO BONITA PARA O ALUNO --- let texto = "📊 Resultado da sua consulta:\n"; if (colsUsuario.length > 0) { texto += "┌" + "─".repeat(colsUsuario.join(" | ").length + 2) + "┐\n"; texto += "│ " + colsUsuario.join(" | ") + " │\n"; texto += "├" + "─".repeat(colsUsuario.join(" | ").length + 2) + "┤\n"; if (valsUsuario.length > 0) { valsUsuario.forEach(row => { texto += "│ " + row.join(" | ") + " │\n"; }); } else { texto += "│ (nenhuma linha) │\n"; } texto += "└" + "─".repeat(colsUsuario.join(" | ").length + 2) + "┘\n"; } else { texto += "(sem colunas)"; } outputEl.textContent = texto; if (iguais) { // ACERTOU! if (!modoVisualizacao && !progresso[indiceFaseAtual].xp) { progresso[indiceFaseAtual] = { xp: true, validado: false, codigo: editorSQL.getValue().trim() }; pontuacaoTotal += fase.pontos || 100; let novoLimite = ultimoLimite; if (novoLimite === undefined || novoLimite === null || novoLimite === -1) { novoLimite = indiceFaseAtual + 1; } else { novoLimite = Math.max(novoLimite, indiceFaseAtual + 1); } if (novoLimite > listaDesafios.length - 1) { novoLimite = listaDesafios.length - 1; } ultimoLimite = novoLimite; await salvarAlunoCloud({ nomeExibicao: alunoAtual, pontuacao: pontuacaoTotal, progresso: progresso, turma: turmaAtual, faseMaximaLiberada: novoLimite }); alert(`🎉 Correto! +${fase.pontos || 100} XP.`); atualizarStatus(); indiceFaseAtual = calcularFaseAtual(); carregarFase(); } else { alert("✅ Sua consulta está correta!"); } } else { // INCORRETO let textoEsperado = "📊 Resultado esperado:\n"; const colsEsp = resultadoGabarito[0].columns; const valsEsp = resultadoGabarito[0].values; if (colsEsp.length > 0) { textoEsperado += "┌" + "─".repeat(colsEsp.join(" | ").length + 2) + "┐\n"; textoEsperado += "│ " + colsEsp.join(" | ") + " │\n"; textoEsperado += "├" + "─".repeat(colsEsp.join(" | ").length + 2) + "┤\n"; if (valsEsp.length > 0) { valsEsp.forEach(row => { textoEsperado += "│ " + row.join(" | ") + " │\n"; }); } else { textoEsperado += "│ (nenhuma linha) │\n"; } textoEsperado += "└" + "─".repeat(colsEsp.join(" | ").length + 2) + "┘\n"; } alert(`❌ Resultado incorreto.\n\nSua saída:\n${texto}\n\nSaída esperada:\n${textoEsperado}`); } } catch (erro) { console.error(erro); alert("Erro ao executar: " + erro.message); } finally { btn.disabled = false; btn.textContent = "⚡ Executar consulta"; } }
